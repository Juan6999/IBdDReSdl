-- Crear tabla SampleData
CREATE TABLE SampleData (
  IntegerValue INTEGER PRIMARY KEY
);

-- Obtener el segundo número entero más alto (forma 1)
SELECT 'Segundo número más alto (LIMIT OFFSET)' AS Metodo;
SELECT IntegerValue 
FROM SampleData 
ORDER BY IntegerValue DESC 
LIMIT 1 OFFSET 1;

-- Obtener el segundo número más alto (forma 2)
SELECT 'Segundo número más alto (MAX con subconsulta)' AS Metodo;
SELECT MAX(IntegerValue)
FROM SampleData
WHERE IntegerValue < (
  SELECT MAX(IntegerValue) FROM SampleData
);

SELECT 'Segundo salario más alto (LIMIT OFFSET)' AS Metodo;
SELECT DISTINCT Salary
FROM Employees
ORDER BY Salary DESC
LIMIT 1 OFFSET 1;

-- Obtener el segundo salario más alto (forma 2)
SELECT 'Segundo salario más alto (MAX con subconsulta)' AS Metodo;
SELECT MAX(Salary)
FROM Employees
WHERE Salary < (
  SELECT MAX(Salary) FROM Employees
);
